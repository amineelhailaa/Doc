	class CompteBancaire{
		private $id;
		private $owner;
		private $solde=0;
		

		public function __construct($id,$owner){
		$this->id = $id;
		$this->owner = $owner;
		}

		public function deposer($montant){
		if($montant<=0) return;
		$this->solde= $this->solde + $montant;
		}
		
		public function retirer($montant){
		if($this->solde<$montant) return;
		$this->solde = $this->solde - $montant;
		return $montant;
		}
}
